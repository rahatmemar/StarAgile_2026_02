pipeline {
    agent any

    tools {
        jdk 'jdk21'
    }

    stages {

        stage('Git Checkout') {
            steps {
                git 'https://github.com/KastroVKiran/my-prime-v2.git'
            }
        }

        stage('Maven Build') {
            steps {
                sh 'mvn clean package'
            }
        }

    }

    post {
        always {
            script {
                def buildStatus = currentBuild.currentResult
                def buildUser = currentBuild.getBuildCauses('hudson.model.Cause$UserIdCause')[0]?.userId ?: 'Github User'

                emailext(
                    subject: "Pipeline ${buildStatus}: ${env.JOB_NAME} #${env.BUILD_NUMBER}",
                    body: """
                        <p>This is a Jenkins CICD pipeline status.</p>
                        <p>Project: ${env.JOB_NAME}</p>
                        <p>Build Number: ${env.BUILD_NUMBER}</p>
                        <p>Build Status: ${buildStatus}</p>
                        <p>Started by: ${buildUser}</p>
                        <p>Build URL: <a href="${env.BUILD_URL}">${env.BUILD_URL}</a></p>
                    """,
                    to: 'kastrokiran@gmail.com',
                    from: 'kastrokiran@gmail.com',
                    replyTo: 'kastrokiran@gmail.com',
                    mimeType: 'text/html'
                )
            }
        }
    }
}

// 🔧 Reusable Slack Notification Function
def sendSlackNotification(status) {

    // Get triggering user
    def user = currentBuild.getBuildCauses('hudson.model.Cause$UserIdCause')[0]?.userId ?: 'GitHub Trigger'

    // Status colors
    def colorMap = [
        'SUCCESS': '#2eb886',   // Green
        'FAILURE': '#e01e5a',   // Red
        'UNSTABLE': '#daa038',  // Yellow
        'ABORTED': '#808080'    // Grey
    ]

    // Emojis for quick visibility
    def emojiMap = [
        'SUCCESS': '🚀',
        'FAILURE': '🔥',
        'UNSTABLE': '⚠️',
        'ABORTED': '⏹️'
    ]

    // Git details (safe fallback handling)
    def branch = env.GIT_BRANCH ?: 'N/A'
    def commit = env.GIT_COMMIT ? env.GIT_COMMIT.take(7) : 'N/A'

    // Build duration
    def duration = currentBuild.durationString.replace(' and counting', '')

    // Final Slack message
    def message = """
${emojiMap[status]} *${status} — Jenkins Pipeline Notification*

*📦 Project:* ${env.JOB_NAME}
*🔢 Build:* #${env.BUILD_NUMBER}
*🌿 Branch:* ${branch}
*🔁 Commit:* ${commit}
*👤 Triggered By:* ${user}
*⏱ Duration:* ${duration}

*🔗 View Build Logs:*  
${env.BUILD_URL}
"""

    slackSend(
        channel: 'C0ANGM3CPFG',
        color: colorMap[status],
        message: message
    )
}

=> Parallel Stages
pipeline {
    agent any

    stages {

        stage('Stage 1: Initialization') {
            steps {
                echo "🔹 Initializing the pipeline..."
                echo "🔹 Setting up environment"
            }
        }

        stage('Stage 2: Build') {
            steps {
                echo "🔨 Building the application..."
                echo "🔨 Compiling source code"
            }
        }

        stage('Stage 3: Test') {
            steps {
                echo "🧪 Running test cases..."
                echo "🧪 Validating application"
            }
        }

        stage('Stage 4: Package') {
            steps {
                echo "📦 Packaging the application..."
                echo "📦 Creating artifact (jar/war)"
            }
        }

        stage('Stage 5: Deploy') {
            steps {
                echo "🚀 Deploying the application..."
                echo "🚀 Deployment completed successfully"
            }
        }

    }

    post {
        always {
            echo "📢 Pipeline execution completed!"
        }

        success {
            echo "✅ Pipeline executed successfully!"
        }

        failure {
            echo "❌ Pipeline failed!"
        }
    }
}
